```python
from skycore.common.events import EventBus, Event

event_bus = EventBus()
```

### 5.1 מודול חיישן

```python
# skycore/sensors/my_sensor.py

from skycore.sensors.base import BaseSensor, SensorData
from skycore.common.interfaces import I2CDevice, SPIDevice

class MySensor(BaseSensor):
    """
    My custom sensor implementation.

    Inherits from BaseSensor and implements required methods.
    """

    # Sensor configuration
    SENSOR_NAME = "my_sensor"
    REQUIRED_CALIBRATION = True
    TYPICAL_RATE_HZ = 100

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)

        # Device interface
        self._i2c = I2CDevice(
            bus=self.config.get('bus', 1),
            address=self.config.get('address', 0x68)
        )

        # Internal state
        self._calibration_offset = [0, 0, 0]

    async def initialize(self) -> bool:
        """Initialize the sensor."""
        try:
            # Read sensor ID
            sensor_id = await self._i2c.read_register(0x00)

            if sensor_id != self.expected_id:
                logger.error(f"Invalid sensor ID: {sensor_id}")
                return False

            # Configure sensor
            await self._configure()

            # Run self-test
            if not await self._self_test():
                return False

            self._initialized = True
            return True

        except Exception as e:
            logger.error(f"Sensor init failed: {e}")
            return False

    async def _configure(self):
        """Configure sensor registers."""
        # Set data rate
        await self._i2c.write_register(0x19, 0x07)  # 1kHz

        # Set scale
        await self._i2c.write_register(0x1B, 0x10)  # ±1000 deg/s

        # Enable sensors
        await self._i2c.write_register(0x6B, 0x10)

    async def _self_test(self) -> bool:
        """Run self-test routine."""
        # Read self-test registers
        st_x = await self._i2c.read_register(0x0E)
        st_y = await self._i2c.read_register(0x0F)

        return (st_x & 0x01) and (st_y & 0x01)

    async def read(self) -> SensorData:
        """Read sensor data."""
        # Read raw values
        raw_x = await self._i2c.read_word(0x1E)
        raw_y = await self._i2c.read_word(0x20)
        raw_z = await self._i2c.read_word(0x22)

        # Apply calibration
        cal_x = raw_x + self._calibration_offset[0]
        cal_y = raw_y + self._calibration_offset[1]
        cal_z = raw_z + self._calibration_offset[2]

        # Convert to physical units
        scale = 0.0174533 / 16.384  # deg/s to rad/s, LSB
        x = cal_x * scale
        y = cal_y * scale
        z = cal_z * scale

        return SensorData(
            sensor=self.SENSOR_NAME,
            timestamp=time.time(),
            values=[x, y, z],
            unit="rad/s",
            quality=self._compute_quality(raw_x, raw_y, raw_z)
        )

    async def calibrate(self, samples: int = 1000) -> bool:
        """Calibrate the sensor."""
        logger.info(f"Calibrating {self.SENSOR_NAME}...")

        offsets = [0, 0, 0]

        for _ in range(samples):
            raw = await self.read()
            offsets[0] += raw.values[0]
            offsets[1] += raw.values[1]
            offsets[2] += raw.values[2]

        # Average
        self._calibration_offset = [o / samples for o in offsets]

        logger.info(f"Calibration complete: {self._calibration_offset}")
        return True
```

### 5.2 מודול בקר

```python
# skycore/control/my_controller.py

from skycore.control.base import BaseController, ControlOutput
import numpy as np

class MyController(BaseController):
    """
    Custom controller implementation.

    Implements PID or MPC control for drone stabilization.
    """

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)

        # Gains
        self.kp = config.get('kp', 1.0)
        self.ki = config.get('ki', 0.1)
        self.kd = config.get('kd', 0.5)

        # State
        self._integral = 0.0
        self._prev_error = 0.0

        # Limits
        self.max_output = config.get('max_output', 1.0)
        self.max_rate = config.get('max_rate', 10.0)

    def compute(self, error: float, dt: float) -> float:
        """
        Compute control output.

        Args:
            error: Current error (setpoint - measured)
            dt: Time step

        Returns:
            Control output in range [-1, 1]
        """
        # Proportional
        p = self.kp * error

        # Integral with anti-windup
        self._integral += error * dt
        self._integral = np.clip(self._integral, -1.0, 1.0)
        i = self.ki * self._integral

        # Derivative with filtering
        if dt > 0:
            derivative = (error - self._prev_error) / dt
            # Simple low-pass filter
            alpha = 0.1
            derivative_filtered = alpha * derivative + (1 - alpha) * self._prev_error
        else:
            derivative_filtered = 0

        d = self.kd * derivative_filtered

        # Total output
        output = p + i + d

        # Rate limiting
        output_rate = (output - self._prev_error) / dt if dt > 0 else 0
        if abs(output_rate) > self.max_rate:
            output = self._prev_error + np.sign(output_rate) * self.max_rate * dt

        # Amplitude limiting
        output = np.clip(output, -self.max_output, self.max_output)

        # Update state
        self._prev_error = error
        self.prev_output = output

        return output

    def reset(self):
        """Reset controller state."""
        self._integral = 0.0
        self._prev_error = 0.0
```

```python
# skycore/navigation/my_planner.py

from skycore.navigation.base import BasePathPlanner, PathPoint
import numpy as np

class MyPathPlanner(BasePathPlanner):
    """
    Custom path planning algorithm.

    Implements RRT* or A* path planning with obstacle avoidance.
    """

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)

        # Planning parameters
        self.max_iterations = config.get('max_iterations', 1000)
        self.step_size = config.get('step_size', 10.0)  # meters
        self.goal_sample_rate = config.get('goal_sample_rate', 0.1)
        self.obstacle_clearance = config.get('obstacle_clearance', 5.0)

        # Internal state
        self._tree = None

    async def plan(
        self,
        start: Tuple[float, float, float],
        end: Tuple[float, float, float],
        obstacles: List[Obstacle]
    ) -> List[PathPoint]:
        """
        Plan a path from start to end.

        Args:
            start: (lat, lon, alt) start position
            end: (lat, lon, alt) end position
            obstacles: List of obstacles to avoid

        Returns:
            List of PathPoints forming the planned path
        """
        # Initialize RRT tree
        self._tree = Tree()
        self._tree.add_node(start)

        # RRT* iterations
        for _ in range(self.max_iterations):
            # Sample random point
            if np.random.random() < self.goal_sample_rate:
                sample = end
            else:
                sample = self._sample_space()

            # Find nearest node
            nearest = self._tree.nearest(sample)

            # Steer towards sample
            new_node = self._steer(nearest, sample)

            # Check collision
            if not self._check_collision(nearest, new_node, obstacles):
                continue

            # Add to tree
            self._tree.add_node(new_node, parent=nearest)

            # Check if goal reached
            if self._distance(new_node, end) < self.step_size:
                self._tree.add_node(end, parent=new_node)
                break

        # Extract path
        path = self._tree.path_to(end)

        # Optimize path
        path = self._smooth_path(path)

        return path

    def _sample_space(self) -> Tuple[float, float, float]:
        """Sample random point in space."""
        return (
            np.random.uniform(32.0, 32.1),
            np.random.uniform(34.7, 34.8),
            np.random.uniform(50, 150)
        )

    def _steer(self, from_node: Node, to_point: Tuple[float, float, float]) -> Node:
        """Steer from node towards point."""
        direction = np.array(to_point) - from_node.position
        distance = np.linalg.norm(direction)

        if distance > self.step_size:
            direction = direction / distance * self.step_size

        new_position = from_node.position + direction

        return Node(position=tuple(new_position))

    def _check_collision(
        self,
        from_node: Node,
        to_node: Node,
        obstacles: List[Obstacle]
    ) -> bool:
        """Check if path segment collides with obstacles."""
        for obstacle in obstacles:
            if obstacle.intersects_line(from_node.position, to_node.position):
                if self._distance_to_obstacle(to_node.position, obstacle) < self.obstacle_clearance:
                    return False
        return True

    def _smooth_path(self, path: List[Node]) -> List[PathPoint]:
        """Smooth path using shortcut method."""
        smoothed = [path[0]]

        i = 0
        while i < len(path) - 1:
            for j in range(len(path) - 1, i, -1):
                if self._is_visible(path[i], path[j]):
                    smoothed.append(path[j])
                    i = j
                    break

        return [PathPoint(
            lat=p.position[0],
            lon=p.position[1],
            alt=p.position[2],
            speed=10.0
        ) for p in smoothed]
```
